create table Users(
  ID int IDENTITY(1,1) PRIMARY KEY,
  FIRST_NAME varchar(250) NOT NULL,
  LAST_NAME varchar(250) NOT NULL,
  EMAIL varchar(250) NOT NULL,
  PASS varchar(250) NOT NULL,
  CONSTRAINT UC_User_Email UNIQUE (EMAIL)
);

create table Category(
  ID int IDENTITY(1,1) PRIMARY KEY,
  CATEGORY_NAME varchar(250) NOT NULL
);

create table Album(
  ID int IDENTITY(1,1) PRIMARY KEY,
  ALBUM_NAME varchar(250) NOT NULL,
  ALBUM_IMG varbinary(max) NOT NULL,
  USER_ID int NOT NULL FOREIGN KEY REFERENCES USERS(ID),
  CATEGORY_ID int NOT NULL FOREIGN KEY REFERENCES CATEGORY(ID)
);

create table Photo(
  ID int IDENTITY(1,1) PRIMARY KEY,
  PHOTO_NAME varchar(250) NOT NULL,
  PHOTO_IMG varbinary(max) NOT NULL,
  ALBUM_ID int NOT NULL FOREIGN KEY REFERENCES ALBUM(ID)
);

create table Comment(
  ID int IDENTITY(1,1) PRIMARY KEY,
  CONTENT varchar(250) NOT NULL,
  USER_ID int NOT NULL FOREIGN KEY REFERENCES USERS(ID),
  ALBUM_ID int NOT NULL FOREIGN KEY REFERENCES ALBUM(ID)
);
